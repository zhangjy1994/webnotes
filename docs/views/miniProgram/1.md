# 微信小程序

## 自定义小程序声明周期
``` js
const oldPage = Page
Page = function(options) {
if(options.onLoad) {
  const oldOnload = options.onLoad
  options.onLoad = function(...args) {
    console.log(1111)
    this.setData({
      start: Date.now()
    })
    oldOnload.apply(this, args)
  }
} else {
  options.onLoad = function(...args) {
    this.setData({
      start: Date.now()
    })
  }
}
oldPage(options)
}
```
